function register() { let username = document.getElementById("reg_username").value; let password = document.getElementById("reg_password").value; fetch("/users/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username, password }) }) .then(response => response.json()) .then(data => { if (data.id) { alert("Регистрация успешна! Теперь войдите в систему."); window.location.href = "/"; } else { alert("Ошибка: " + data.detail); } }); }
